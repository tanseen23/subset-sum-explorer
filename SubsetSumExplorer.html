// Pseudocode
ALGORITHM SubsetSumBacktrack(S, n, T, index, currentSubset, currentSum)

  // Base Case: Target Achieved
  IF currentSum == T THEN
    OUTPUT currentSubset
    RETURN
  END IF

  // Pruning: Overshoot or exhausted elements
  IF currentSum > T OR index >= n THEN
    RETURN // Prune this branch
  END IF

  // Decision 1: INCLUDE S[index]
  currentSubset.APPEND(S[index])
  SubsetSumBacktrack(S, n, T, index+1, currentSubset, currentSum + S[index])

  // Decision 2: EXCLUDE S[index] (Backtrack)
  currentSubset.REMOVE_LAST()
  SubsetSumBacktrack(S, n, T, index+1, currentSubset, currentSum)

END ALGORITHM
